## Exemplo pratico do Result no tratamento de erros
---

```rust 

use std::fs;
use std::num::ParseIntError;

// Funçao que pode falhar - retorna Result
fn dividir(a: &str, b: &str) -> Result<f64, String> {
    // Tenta converter strings para nÃºmeros
    let num_a: f64 = a.parse().map_err(|_| "Primeiro numero invalido".to_string())?;
    let num_b: f64 = b.parse().map_err(|_| "Segundo numero invalido".to_string())?;
    
    // Verifica divisao por zero
    if num_b == 0.0 {
        return Err("Divisao por zero nao permitida".to_string());
    }
    
    Ok(num_a / num_b)
}

// Funçao que le arquivo e pode falhar
fn ler_configuracao(arquivo: &str) -> Result<String, String> {
    fs::read_to_string(arquivo)
        .map_err(|_| format!("Erro ao ler arquivo: {}", arquivo))
}

fn main() {
    // === TRATAMENTO COM MATCH (equivalente ao try/catch) ===
    println!("=== Usando match (como try/catch) ===");
    
    match dividir("10", "2") {
        Ok(resultado) => println!("Resultado: {}", resultado),
        Err(erro) => println!("Erro: {}", erro),
    }
    
    match dividir("10", "0") {
        Ok(resultado) => println!("Resultado: {}", resultado),
        Err(erro) => println!("Erro: {}", erro),
    }
    
    match dividir("abc", "2") {
        Ok(resultado) => println!("Resultado: {}", resultado),
        Err(erro) => println!("Erro: {}", erro),
    }
    
    // === TRATAMENTO COM IF LET ===
    println!("\n=== Usando if let ===");
    
    if let Ok(resultado) = dividir("15", "3") {
        println!("Deu certo: {}", resultado);
    } else {
        println!("Algo deu errado");
    }
    
    // === UNWRAP_OR (valor padrÃ£o se der erro) ===
    println!("\n=== Usando unwrap_or ===");
    
    let resultado1 = dividir("20", "4").unwrap_or(0.0);
    let resultado2 = dividir("20", "0").unwrap_or(0.0);
    
    println!("Resultado 1: {}", resultado1); // 5.0
    println!("Resultado 2: {}", resultado2); // 0.0 (valor padrÃ£o)
    
    // === PROPAGAÃÃO DE ERRO COM ? ===
    println!("\n=== Testando leitura de arquivo ===");
    
    match ler_configuracao("config.txt") {
        Ok(conteudo) => println!("Arquivo lido: {}", conteudo),
        Err(erro) => println!("Erro: {}", erro),
    }
}
```
